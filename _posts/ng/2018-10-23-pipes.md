## Custom Pipes

```js
/**
 *  customPipe.pipe.ts
 */
import { Pipe, PipeTransform } from '@angular/core';

@Pipe({
  name: 'pipeName'
})
export class CustomPipe implements PipeTransform {
  transform(value) {
    ...
	return formattedValue;
  }
}

/**
 *  app.module.ts
 */
...
import { CustomPipe } from './customPipe.pipe';
...
@NgModule({
  ...,
  declarations: [
    ...,
	CustomPipe
  ]
})
```

...
